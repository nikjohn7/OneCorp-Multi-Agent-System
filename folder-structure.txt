    source-of-truth/
      EOI_John_JaneSmith.pdf
        # Original EOI PDF (unchanged) — canonical source of truth for deal fields.

    contracts/
      CONTRACT_V1.pdf
        # Incorrect contract PDF — has 5 known discrepancies vs EOI.

        # Corrected contract PDF — should match EOI exactly.

    emails/
      incoming/
        01_eoi_signed.txt
          # Raw email body for "EOI Signed" — triggers deal creation.
        02_contract_v1_received.txt
          # Email from vendor sending incorrect contract (V1) as attachment.
        02b_contract_v2_received.txt
          # Email from vendor sending corrected contract (V2) as attachment.
        04_solicitor_approved.txt
          # Email from solicitor confirming approval and signing appointment time.
        06_docusign_please_sign.txt
          # DocuSign "please sign" email to buyers — triggers DOCUSIGN_RELEASED.
        07_buyer_signed.txt
          # DocuSign "buyer signed" email — triggers BUYER_SIGNED state.
        08_contract_executed.txt
          # DocuSign "completed/executed" email — triggers EXECUTED state.

      templates/
        03_contract_to_solicitor.txt
          # OUTGOING TEMPLATE — contract-to-solicitor email format.
        05_vendor_release_request.txt
          # OUTGOING TEMPLATE — "please release via DocuSign" email format.

    emails_manifest.json
